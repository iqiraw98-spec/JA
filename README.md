// فتح الكاميرا
async function openCamera() {
  document.getElementById('cameraModal').classList.add('open');
  document.getElementById('scanResult').style.display = 'none';
  document.getElementById('cameraStatus').textContent = 'جاري تشغيل الكاميرا...';

  try {
    // الحصول على قائمة الكاميرات
    availableCameras = await Html5Qrcode.getCameras();
    if (!availableCameras || availableCameras.length === 0) {
      document.getElementById('cameraStatus').textContent = '❌ لا توجد كاميرا متاحة';
      return;
    }

    // تفضيل الكاميرا الخلفية
    const backCam = availableCameras.find(c =>
      c.label.toLowerCase().includes('back') ||
      c.label.toLowerCase().includes('rear') ||
      c.label.toLowerCase().includes('environment')
    );
    currentCameraIndex = backCam ? availableCameras.indexOf(backCam) : 0;

    startScanner(availableCameras[currentCameraIndex].id);

  } catch (err) {
    document.getElementById('cameraStatus').textContent = '❌ خطأ: ' + err.message;
    console.error(err);
  }
}

function startScanner(cameraId) {
  if (html5QrCode) {
    html5QrCode.stop().catch(() => {}).finally(() => {
      html5QrCode = null;
      _startQrCode(cameraId);
    });
  } else {
    _startQrCode(cameraId);
  }
}

function _startQrCode(cameraId) {

  html5QrCode.start(
    cameraId,
    {
      fps: 15,
      qrbox: { width: 260, height: 160 },
      aspectRatio: 1.5,
      formatsToSupport: [
        Html5QrcodeSupportedFormats.EAN_13,
        Html5QrcodeSupportedFormats.EAN_8,
        Html5QrcodeSupportedFormats.CODE_128,
        Html5QrcodeSupportedFormats.CODE_39,
        Html5QrcodeSupportedFormats.UPC_A,
        Html5QrcodeSupportedFormats.UPC_E,
        Html5QrcodeSupportedFormats.QR_CODE,
        Html5QrcodeSupportedFormats.DATA_MATRIX,
      ]
    },
    (decodedText) => {
      // منع التكرار السريع
      if (scanCooldown || decodedText === lastScanned) return;
      lastScanned = decodedText;
      scanCooldown = true;
      setTimeout(() => { scanCooldown = false; lastScanned = ''; }, 2000);

      // اهتزاز على الجوال
      if (navigator.vibrate) navigator.vibrate(100);

      // عرض النتيجة
      document.getElementById('scanResultText').textContent = decodedText;
      const p = products.find(x => x.barcode === decodedText);
      const resultEl = document.getElementById('scanResult');
      const nameEl = document.getElementById('scanProductName');
      resultEl.style.display = 'block';

      if (p) {
        nameEl.textContent = '✅ ' + p.name + ' — ' + p.price.toFixed(2) + ' ' + settings.currency;
        nameEl.style.color = 'var(--success)';
        addToCart(p.id);
        showToast('✅ تم مسح: ' + p.name, 'success');
        // إغلاق تلقائي بعد ثانية
        setTimeout(() => closeCamera(), 1200);
      } else {
        nameEl.textContent = '⚠️ الباركود غير موجود في قاعدة البيانات';
        nameEl.style.color = 'var(--warning)';
        // نسخ الباركود لحقل البحث
        document.getElementById('searchInput').value = decodedText;
      }
    },
    () => {} // خطأ المسح - تجاهل
  ).catch(err => {
    document.getElementById('cameraStatus').textContent = '❌ تعذّر فتح الكاميرا. تأكد من منح الإذن.';
    console.error(err);
  });
}

let barcodeBuffer = '';
let barcodeTimer = null;
document.addEventListener('keydown', (e) => {
  // تجاهل إذا كان المستخدم يكتب في حقل نص
  const tag = document.activeElement.tagName;
  if (tag === 'INPUT' || tag === 'TEXTAREA' || tag === 'SELECT') return;

  if (e.key === 'Enter') {
    if (barcodeBuffer.length >= 4) {
      const p = products.find(x => x.barcode === barcodeBuffer);
      if (p) {
        addToCart(p.id);
        showToast('📷 تم المسح: ' + p.name, 'success');
      } else {
        document.getElementById('searchInput').value = barcodeBuffer;
        showToast('⚠️ باركود غير موجود: ' + barcodeBuffer, 'error');
      }
    }
    barcodeBuffer = '';
    if (barcodeTimer) clearTimeout(barcodeTimer);
  } else if (e.key.length === 1) {
    barcodeBuffer += e.key;
    if (barcodeTimer) clearTimeout(barcodeTimer);
    barcodeTimer = setTimeout(() => { barcodeBuffer = ''; }, 100);
  }
});

function addToCart(id) {
  const p = products.find(x => x.id === id);
  if (!p || p.stock <= 0) { showToast('❌ المنتج غير متوفر', 'error'); return; }
  const existing = cart.find(c => c.id === id);
  if (existing) {
    if (existing.qty >= p.stock) { showToast('⚠️ لا يوجد مخزون كافٍ', 'error'); return; }
    existing.qty++;
  } else {
    cart.push({ id, qty: 1, name: p.name, price: p.price });
  }
  renderCart();
  showToast('✅ تمت الإضافة: ' + p.name, 'success');
}

function removeFromCart(id) {
  cart = cart.filter(c => c.id !== id);
  renderCart();
}

function changeQty(id, delta) {
  const item = cart.find(c => c.id === id);
  const p = products.find(x => x.id === id);
  if (!item) return;
  item.qty += delta;
  if (item.qty <= 0) { removeFromCart(id); return; }
  if (item.qty > p.stock) { item.qty = p.stock; showToast('⚠️ الكمية المتاحة: ' + p.stock, 'error'); }
  renderCart();
}

function clearCart() {
  cart = [];
  renderCart();
  document.getElementById('discountInput').value = '';
  document.getElementById('discountAmount').value = '';
}

function renderCart() {
  const el = document.getElementById('cartItems');
  document.getElementById('cartCount').textContent = cart.reduce((s, i) => s + i.qty, 0);
  if (cart.length === 0) {
    el.innerHTML = `<div class="cart-empty"><div class="cart-empty-icon">🛒</div><div>اختر منتجاً للبدء</div></div>`;
    updateTotals();
    return;
  }
  el.innerHTML = cart.map(item => `
    <div class="cart-item">
      <div class="ci-name">${item.name}</div>
      <div class="ci-qty">
        <button class="qty-btn" onclick="changeQty(${item.id}, -1)">−</button>
        <span class="qty-val">${item.qty}</span>
        <button class="qty-btn" onclick="changeQty(${item.id}, 1)">+</button>
      </div>
      <div class="ci-price">${(item.price * item.qty).toFixed(2)}</div>
      <button class="ci-del" onclick="removeFromCart(${item.id})">🗑</button>
    </div>`).join('');
  updateTotals();
}

function updateTotals() {
  const subtotal = cart.reduce((s, i) => s + i.price * i.qty, 0);
  const discPct = parseFloat(document.getElementById('discountInput').value) || 0;
  const discAmt = parseFloat(document.getElementById('discountAmount').value) || 0;
  const discount = Math.min(subtotal, subtotal * (discPct / 100) + discAmt);
  const total = subtotal - discount;
  document.getElementById('subtotalDisplay').textContent = subtotal.toFixed(2) + ' ' + settings.currency;
  document.getElementById('discountDisplay').textContent = discount.toFixed(2) + ' ' + settings.currency;
  document.getElementById('totalDisplay').textContent = total.toFixed(2) + ' ' + settings.currency;
}

function getTotal() {
  const subtotal = cart.reduce((s, i) => s + i.price * i.qty, 0);
  const discPct = parseFloat(document.getElementById('discountInput').value) || 0;
  const discAmt = parseFloat(document.getElementById('discountAmount').value) || 0;
  const discount = Math.min(subtotal, subtotal * (discPct / 100) + discAmt);
  return subtotal - discount;
}

function openPayment(method) {
  if (cart.length === 0) { showToast('🛒 السلة فارغة!', 'error'); return; }
  currentPaymentMethod = method;
  const total = getTotal();
  document.getElementById('paymentTitle').textContent = method === 'cash' ? '💵 الدفع النقدي' : '💳 الدفع بالبطاقة';
  document.getElementById('payDue').textContent = total.toFixed(2) + ' ' + settings.currency;
  document.getElementById('payReceived').value = '';
  document.getElementById('changeDisplay').style.display = 'none';
  buildNumpad();
  if (method === 'card') {
    document.getElementById('payReceived').value = total.toFixed(2);
    calcChange();
  }
  document.getElementById('paymentModal').classList.add('open');
}

function buildNumpad() {
  const np = document.getElementById('numpad');
  const keys = ['1','2','3','4','5','6','7','8','9','','0','.'];
  np.innerHTML = keys.map((k, i) => {
    if (k === '') return `<button class="num-btn clear" onclick="clearPay()">⌫</button>`;
    const cls = k === '0' ? '' : '';
    return `<button class="num-btn ${cls}" onclick="appendNum('${k}')">${k}</button>`;
  }).join('');
}

function appendNum(n) {
  const inp = document.getElementById('payReceived');
  if (n === '.' && inp.value.includes('.')) return;
  inp.value += n;
  calcChange();
}

function clearPay() {
  const inp = document.getElementById('payReceived');
  inp.value = inp.value.slice(0, -1);
  calcChange();
}

function calcChange() {
  const received = parseFloat(document.getElementById('payReceived').value) || 0;
  const due = getTotal();
  const change = received - due;
  const cd = document.getElementById('changeDisplay');
  if (received > 0) {
    cd.style.display = 'block';
    document.getElementById('changeAmount').textContent = (change >= 0 ? change : 0).toFixed(2) + ' ' + settings.currency;
    cd.style.borderColor = change >= 0 ? 'var(--success)' : 'var(--danger)';
    document.getElementById('changeAmount').style.color = change >= 0 ? 'var(--success)' : 'var(--danger)';
  } else {
    cd.style.display = 'none';
  }
}

function completeSale() {
  const received = parseFloat(document.getElementById('payReceived').value) || 0;
  const total = getTotal();
  if (currentPaymentMethod === 'cash' && received < total) {
    showToast('⚠️ المبلغ المدفوع غير كافٍ', 'error');
    return;
  }
  // Deduct stock
  cart.forEach(item => {
    const p = products.find(x => x.id === item.id);
    if (p) p.stock -= item.qty;
  });
  // Save sale
  const sale = {
    id: Date.now(),
    date: new Date().toISOString(),
    items: [...cart],
    subtotal: cart.reduce((s, i) => s + i.price * i.qty, 0),
    discount: cart.reduce((s, i) => s + i.price * i.qty, 0) - total,
    total,
    method: currentPaymentMethod,
    received: currentPaymentMethod === 'cash' ? received : total,
    change: currentPaymentMethod === 'cash' ? received - total : 0
  };
  sales.push(sale);
  saveData();
  closeModal('paymentModal');
  showReceipt(sale);
  clearCart();
  renderProductsGrid();
}

function renderInventory() {
  const q = (document.getElementById('invSearch').value || '').toLowerCase();
  const cat = document.getElementById('invCatFilter').value;
  const cats = getCategories();
  const cf = document.getElementById('invCatFilter');
  const currentVal = cf.value;
  cf.innerHTML = '<option value="">كل الفئات</option>' + cats.map(c => `<option ${c === currentVal ? 'selected' : ''} value="${c}">${c}</option>`).join('');
  cf.value = currentVal;

  let filtered = products.filter(p => {
    const matchQ = !q || p.name.toLowerCase().includes(q) || p.barcode.includes(q) || (p.scientific || '').toLowerCase().includes(q);
    const matchCat = !cat || p.category === cat;
    return matchQ && matchCat;
  });

  const tbody = document.getElementById('invBody');
  tbody.innerHTML = filtered.map(p => {
    const status = p.stock <= 0 ? '<span class="badge badge-out">نفذت</span>' :
      p.stock <= settings.lowStockThreshold ? '<span class="badge badge-low">منخفض</span>' :
      '<span class="badge badge-ok">متوفر</span>';

    const expiry = p.expiry ? checkExpiry(p.expiry) : '<span class="expiry-badge expiry-ok">—</span>';

    return `<tr>
      <td><span style="font-family:'IBM Plex Mono',monospace;font-size:0.75rem;">${p.barcode}</span></td>
      <td><strong>${p.name}</strong><br><span style="font-size:0.72rem;color:var(--text2);">${p.scientific || ''}</span></td>
      <td>${p.category}</td>
      <td><strong>${p.price.toFixed(2)} ${settings.currency}</strong><br><span style="font-size:0.72rem;color:var(--text2);">تكلفة: ${(p.cost || 0).toFixed(2)}</span></td>
      <td><strong>${p.stock}</strong> ${p.unit || ''}</td>
      <td>${expiry}</td>
      <td>${status}</td>
      <td>
        <div class="action-btns">
          <button class="act-btn" onclick="openEditProduct(${p.id})">✏️ تعديل</button>
          <button class="act-btn" onclick="addStock(${p.id})">➕ إضافة</button>
          <button class="act-btn danger" onclick="deleteProduct(${p.id})">🗑</button>
        </div>
      </td>
    </tr>`;
  }).join('');
}

function checkExpiry(expiry) {
  const now = new Date();
  const exp = new Date(expiry + '-01');
  const diff = (exp - now) / (1000 * 60 * 60 * 24);
  if (diff < 0) return `<span class="expiry-badge expiry-exp">منتهي ❌ (${expiry})</span>`;
  if (diff <= settings.expiryWarnDays) return `<span class="expiry-badge expiry-soon">⚠️ ${expiry}</span>`;
  return `<span class="expiry-badge expiry-ok">${expiry}</span>`;
}

function saveProduct() {
  const name = document.getElementById('pName').value.trim();
  const barcode = document.getElementById('pBarcode').value.trim();
  const price = parseFloat(document.getElementById('pPrice').value);
  const stock = parseInt(document.getElementById('pStock').value);
  if (!name || !barcode || isNaN(price) || isNaN(stock)) {
    showToast('⚠️ يرجى ملء جميع الحقول المطلوبة', 'error'); return;
  }
  const data = {
    name, barcode,
    scientific: document.getElementById('pScientific').value.trim(),
    category: document.getElementById('pCategory').value,
    price, cost: parseFloat(document.getElementById('pCost').value) || 0,
    stock,
    unit: document.getElementById('pUnit').value,
    expiry: document.getElementById('pExpiry').value,
    manufacturer: document.getElementById('pManufacturer').value.trim(),
    icon: document.getElementById('pIcon').value.trim() || '💊'
  };
  if (editingProductId) {
    const idx = products.findIndex(x => x.id === editingProductId);
    products[idx] = { ...products[idx], ...data };
    showToast('✅ تم تحديث المنتج', 'success');
  } else {
    data.id = Date.now();
    products.push(data);
    showToast('✅ تمت إضافة المنتج', 'success');
  }
  saveData();
  closeModal('productModal');
  renderInventory();
  renderProductsGrid();
  renderCategoryFilters();
}

function addStock(id) {
  const p = products.find(x => x.id === id);
  const qty = parseInt(prompt(`إضافة كمية لـ "${p.name}" (الكمية الحالية: ${p.stock})`));
  if (!qty || qty < 1) return;
  p.stock += qty;
  saveData();
  renderInventory();
  renderProductsGrid();
  showToast(`✅ تمت إضافة ${qty} وحدة لـ ${p.name}`, 'success');
}

function deleteProduct(id) {
  if (!confirm('هل تريد حذف هذا المنتج؟')) return;
  products = products.filter(x => x.id !== id);
  saveData();
  renderInventory();
  renderProductsGrid();
  renderCategoryFilters();
  showToast('🗑 تم حذف المنتج', 'success');
}

function renderReports() {
  const todaySales = sales.filter(s => new Date(s.date).toDateString() === new Date().toDateString());

  const totalRevenue = sales.reduce((s, x) => s + x.total, 0);

  const outOfStock = products.filter(p => p.stock <= 0).length;

  `;

  // Top products
  const soldMap = {};
  sales.forEach(s => s.items.forEach(i => {
    soldMap[i.name] = (soldMap[i.name] || 0) + i.qty;
  }));
  const top = Object.entries(soldMap).sort((a, b) => b[1] - a[1]).slice(0, 6);
  const maxSold = top[0]?.[1] || 1;
  document.getElementById('topProducts').innerHTML = top.length ? top.map(([name, qty]) => `
    <div class="sale-bar-wrap">
      <div class="sale-bar-label"><span>${name}</span><span>${qty} وحدة</span></div>
      <div class="sale-bar"><div class="sale-bar-fill" style="width:${(qty / maxSold * 100)}%"></div></div>
    </div>`).join('') : '<div style="color:var(--text2);font-size:0.85rem;">لا توجد مبيعات بعد</div>';

  // Stock alerts
  const alerts = products.filter(p => p.stock <= settings.lowStockThreshold);
  document.getElementById('stockAlerts').innerHTML = alerts.length ? alerts.slice(0, 8).map(p => `
    <div class="sale-row">
      <span>${p.name}</span>
      <span class="badge ${p.stock <= 0 ? 'badge-out' : 'badge-low'}">${p.stock <= 0 ? 'نفذت' : p.stock + ' متبقٍ'}</span>
    </div>`).join('') : '<div style="color:var(--success);font-size:0.85rem;">✅ جميع المنتجات متوفرة</div>';

  // Recent sales
  const recent = [...sales].reverse().slice(0, 8);
  document.getElementById('recentSales').innerHTML = recent.length ? recent.map(s => `
    <div class="sale-row">
      <span style="font-size:0.78rem;">${new Date(s.date).toLocaleString('ar')}</span>
      <span style="color:var(--success);font-weight:700;">${s.total.toFixed(2)} ${settings.currency}</span>
    </div>`).join('') : '<div style="color:var(--text2);font-size:0.85rem;">لا توجد مبيعات</div>';

  // Expiry
  const expiring = products.filter(p => {
    if (!p.expiry) return false;
    const exp = new Date(p.expiry + '-01');
    const diff = (exp - new Date()) / (1000 * 60 * 60 * 24);
    return diff <= 90;
  }).sort((a, b) => new Date(a.expiry) - new Date(b.expiry));
  document.getElementById('expiryReport').innerHTML = expiring.length ? expiring.slice(0, 8).map(p => `
    <div class="sale-row">
      <span>${p.name}</span>
      ${checkExpiry(p.expiry)}
    </div>`).join('') : '<div style="color:var(--success);font-size:0.85rem;">✅ لا توجد منتجات قريبة الانتهاء</div>';
}

  settings.pharmLicense = document.getElementById('pharmLicense').value;
  settings.pharmAddress = document.getElementById('pharmAddress').value;
  settings.pharmPhone = document.getElementById('pharmPhone').value;

  showToast('🔄 تمت إعادة الضبط', 'success');
  renderProductsGrid();
  renderCategoryFilters();
}

function exportAll() {
  const data = JSON.stringify({ products, sales, settings }, null, 2);
  const blob = new Blob([data], { type: 'application/json' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = 'pharmacy_backup_' + new Date().toISOString().slice(0,10) + '.json';
  a.click();
  showToast('💾 تم تصدير البيانات', 'success');
}

function installApp() {
  if (deferredPrompt) {
    deferredPrompt.prompt();
    deferredPrompt.userChoice.then(choice => {
      if (choice.outcome === 'accepted') showToast('✅ جاري تثبيت التطبيق...', 'success');
      deferredPrompt = null;
      const btn = document.getElementById('installBtn');
      if (btn) btn.style.display = 'none';
    });
  }
}
